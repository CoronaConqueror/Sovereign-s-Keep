#version 330 core

out vec4 FragColor;

in vec3 ourColor;
in vec2 TexCoord;
flat in int gui;

uniform sampler2D ourTexture;

void main(){

	if(gui == 1){
	FragColor = vec4(1.0,0.0,0.0,1.0);
	}
	else if(gui == 2){
	FragColor = vec4(0.0,0.0,1.0,1.0);
	}
	else if(gui == -1)
	{
	FragColor = texture(ourTexture, TexCoord) * vec4(15.0, 0.0, 0.0, 1.0);
	}
	else
	{
	FragColor = texture(ourTexture, TexCoord);
	}
	
}
